import type { PrivacySettings } from '../types/types.js';

const store = new Map<string, PrivacySettings>();

export function get(jid: string): PrivacySettings {
  return store.get(jid) ?? {};
}

export function update(jid: string, patch: Partial<PrivacySettings>): PrivacySettings {
  const current = store.get(jid) ?? {};
  const updated = { ...current, ...patch };
  store.set(jid, updated);
  return updated;
}
